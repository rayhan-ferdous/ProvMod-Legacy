Global

class __declspec(dllexport) Global {

public:

static long Freq; - daily observation frequency, e.g. 24 for hourly data derived from first observation file opened.

static long DTindx; - current observation index during model run.

static long DTmin; - minimum possible run index calculated from CRHM run start time.

static long DTmax; - maximum possible run index calculated from CRHM run end time.

static long IndxMin; - minimum possible observation index.

static long IndxMax; - maximum possible observation index.

static long maxobs; - maximum observation dimension.

static long maxlay; - maximum soil layer dimension.

static long maxhru; - maximum HRU dimension.

static double Interval; - run time step as fraction of day(1/Freq).

static double DTstart; - CRHM run start time from user.

static double DTend; - CRHM run end time from user.

static TDateTime DTnow; - CRHM absolute current time during run.

static TBuild BuildFlag; - BUILD/DECL/INIT.

static MapVar MapVars; - map object containing current variables instantiated in model DECL phase. <module name + varName, varAddress>

static MapPar MapPars; - map object containing current parameters instantiated in model DECL phase. <module name+ parName, parAddress>

static MapDim MapDims; - map object containing current dimensions instantiated in model DECL phase. <dimension name, value>

static Mapstr Mapgetvar; - map object containing current getvar's instantiated in model BUILD phase. <module name, module + varName>

static Mapstr Mapputvar; - map object containing current putvar's instantiated in model BUILD phase. <module name, module + varName>

static Mapstr Mapreadvar; - map object containing current readvar's instantiated in model BUILD phase. <module name, varName>

static Mapstr Mapdeclvar; - map object containing current declvar's instantiated in model BUILD phase. <module name, varName>

static Mapstr Mapdeclpar; - map object containing current declpar's instantiated in model BUILD phase. <module name, varName>

static TStringList *AllModulesList; - modules defined in current CRHM model implementation.

static TStringList *OurModulesList; - modules used in current model project.

static TStringList *AllModelsList; - models defined in current CRHM model implementation.

static TStringList *ModelModulesList; // modules used in above models in CommaText.

static TStringList *OurHelpListList; // list of help file names by DLL.

static HWND crhmMain; - handle to send messages to Main.

static HWND crhmLog; - handle to send messages to log and user screens.

};
